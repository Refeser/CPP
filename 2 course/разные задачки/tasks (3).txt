Дан текст произвольной длины, оканчивающийся символом <<;>>. Проверить, есть ли в тексте скобки.

#include <iostream>
#include <conio.h>
using namespace std;
int main()
{
    setlocale (LC_ALL, "Rus");
    int k=0;
    char s;
    cout << "Введите текст:" << endl;
    while((s=cin.get())!=';')
    {
        if((s=='(')||(s==')'))
           k=k++;
    }
    if (k>0)
        cout<<"В предложении есть скобки"<<endl;
    else
        cout<<"В предложении нет скобок"<<endl;
    getch();
    return 0;
}

 Дана строка символов. Признак конца –символ ‘\n’ (перевод строки). Необходимо Определить наибольшее число подряд идущих одинаковых символов.

#include <iostream>
#include <conio.h>

using namespace std;

int main()
{
    setlocale (LC_ALL, "Rus");
    int k=1,Max;
    char s,a;
    cout << "Введите текст:" << endl;
    s=cin.get();
    a=s;
    Max=1;
    while((s=cin.get())!='\n')
    {
        if (s==a)
        {
            k++;
            a=s;
        }
        else
        {
            if(k>Max)
                Max=k;
            k=1;
            a=s;
        }
    }
    cout<<"max кол-во подряд идущих символов="<<Max<<endl;
    getch();
    return 0;
}
