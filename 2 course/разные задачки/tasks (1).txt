Дана строка символов. Признак конца строки — символ «\n» (перевод строки). Строка состоит из слов, разделенных пробелами.
Вывести самое короткое слово и его длину.
#include <iostream>
#include <stdio.h>
#define N 100
using namespace std;

int main()
{
    setlocale(LC_ALL,"Rus");
    int i=0, sltek=0,slmin=0;
    char sl;
    char S[N], SMin[N];
    cout<<"Введите текст"<<endl;
    sl=cin.get();
    while(sl!='\n')
    {
        while(sl==' '&&sl!='\n')
        {
            sl=cin.get();
        }
        while(sl!=' '&&sl!='\n')
        {
            S[i++]=sl;
            sltek++;
            sl=cin.get();
        }
        if(slmin==0)
        {
            slmin=sltek;
            for(i=0;i<sltek;i++)
                SMin[i]=S[i];
        }
        if(sltek<slmin)
        {
            for(i=0;i<sltek;i++)
                SMin[i]=S[i];
            slmin=sltek;
        }
        i=0;
        sltek=0;
    }
    SMin[slmin]='\0';
    cout<<"В самом маленьком слове "<<slmin<<" букв"<<endl<<"Вот это слово:"<<SMin<<endl;
    getchar();
    return 0;
}

{
	  int n,i,c[N],k,j,min;
	  cout <<"Vvedite n<=10:"<<endl;
	  cin>>n;
	  for (i=0;i<n;i++)
	  {
		  cin>>k;
		  c[i]=k;
	  }
	  i=0;
	  min=c[i];
	  j=n;
	  while (j!=0)
	  {
		  for(i=0;i<j;i++)
		  {
			  if(c[i]<c[j])
			  {
				  min=c[i];
				  c[i]=c[j];

			  }
		  }
		  j--;
