Даны натуральные числа n, A1,A2,…,An. Определить количество членов Ak последовательности A1,A2,…,An имеющих четные порядковые номера и являющихся нечетными числами.

#include <iostream>
#include <conio.h>

using namespace std;

int main()
{
   setlocale(LC_ALL,"Rus");
   int n,i=1,k=0,a;
   cout<<"введите количество чисел"<<endl;
   cin>>n;

   for (;i<=n;i++)
   {
       cin>>a;
       if ((i%2==0)&&(a%2!=0))
       {
           k=k+1;
       }
   }

   cout<<"кол-во нечетных чисел с четным порядковым номером = "<<k<<endl;

    getch();
    return 0; 
}

Даны натуральное число n, действительные числа A1,A2,…,An. Получить Удвоенную сумму всех положительных членов последовательности A1,A2,…,An.

#include <iostream>
#include <conio.h>

using namespace std;

int main()
{
   setlocale(LC_ALL,"Rus");
   int n,i=1,s=0,a;
   cout<<"введите количество чисел"<<endl;
   cin>>n;

   for (;i<=n;i++)
   {
       cin>>a;
       if (a>0)
        s=s+a;
   }

   cout<<"удвоенная сумма положительных чисел:"<<2*s<<endl;

    getch();
    return 0;
}

Даны натуральные числа А1,А2,… Признак конца последовательности 0. Определить min(A1+A2,A2+A3,…).

#include <iostream>
#include <conio.h>

using namespace std;

int main()
{
   setlocale(LC_ALL,"Rus");
   int i=1,a,p,minimum;
   cout<<"chisla:"<<endl;
   cin>>a;

   while (a!=0)
   {
       cin>>p;
       if(p==0)
        break;
       if(i==1)
       {
           i++;
           minimum=a+p;
       }
       else
       {
           if (a+p<minimum)
            minimum=a+p;
           i++;
           a=p;
       }
   }

   cout<<"minimum:"<<minimum<<endl;

    getch();
    return 0;
}

Даны натуральные числа А1,А2,… Признак конца последовательности-число 999. Необходимо определить сумму положительных и произведение отрицательных чисел последовательности.

#include <iostream>
#include <conio.h>

using namespace std;

int main()
{
   setlocale(LC_ALL,"Rus");
   int a,s=0,p=1;
   cout<<"chisla:"<<endl;
   cin>>a;

   while (a!=999)
   {
       if(a>0)
        s=s+a;
       else
       {
           if(a<0)
            p=p*a;
       }
       cin>>a;
   }

   cout<<"сумма положительных: "<<s<<endl<<"произведение отрицательных: "<<p<<endl;

    getch();
    return 0;
}
