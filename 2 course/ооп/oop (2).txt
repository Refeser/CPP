/*Выводит истек ли срок годности и количество товара. Срок годности вводится с клавиатуры*/
#include <string>
#include <iostream>
using namespace std;

class Tovar
{
private:
	int kolvo, srok_godnosti_god, srok_vipusk;
	string imya;
public:
	void vvod(int kkolvo, int srok)
	{
		kolvo = kkolvo;
		srok_godnosti_god = srok;
	}
	void tovar(string name)
	{
		imya = name;
	}
	void godnost()
	{
		if (srok_godnosti_god == 2018)
			cout << "Срок годности " << imya << " истекает в этом году. Их " << kolvo << " штук" << endl;
		else
			if (srok_godnosti_god>2018)
				cout << "Срок годности " << imya << " не истекает в этом году. Их " << kolvo << " штук" << endl;
			else
				cout << "Срок годности " << imya << " истек. Их " << kolvo << " штук" << endl;
	}
};

	hleb.vvod(50, srok);
	hleb.tovar("хлеба");
	hleb.godnost();
	cout << "---------------------------------------------------------------------" << endl;
	Tovar sir;

	cout << "---------------------------------------------------------------------" << endl;

	baton.vvod(62, srok);
	baton.tovar("батона");
	baton.godnost();
	cout << "---------------------------------------------------------------------" << endl;
	Tovar pech;
